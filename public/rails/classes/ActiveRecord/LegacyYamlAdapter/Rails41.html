def self.convert(klass, coder)
  attributes = klass.attributes_builder
    .build_from_database(coder["attributes"])
  new_record = coder["attributes"][klass.primary_key].blank?

  {
    "attributes" => attributes,
    "new_record" => new_record,
  }
end
