def initialize
  super 'signin', 'Sign in to any gemcutter-compatible host. '\
        'It defaults to https://rubygems.org'

  add_option('--host HOST', 'Push to another gemcutter-compatible host') do |value, options|
    options[:host] = value
  end

  add_otp_option
end
