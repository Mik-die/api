# File ruby/ext/fiddle/lib/fiddle/import.rb, line 26
def sym(symbol)
  @handlers.each{|handle|
    if( handle )
      begin
        addr = handle.sym(symbol)
        return addr
      rescue DLError
      end
    end
  }
  return nil
end
