static VALUE
syntax_error_initialize(int argc, VALUE *argv, VALUE self)
{
    VALUE mesg;
    if (argc == 0) {
        mesg = rb_fstring_lit("compile error");
        argc = 1;
        argv = &mesg;
    }
    return rb_call_super(argc, argv);
