def each(rs = nil)
  if @pattern[0] == ?/
    @files = Dir[@pattern]
  else
    prefix = @shell.pwd+"/"
    @files = Dir[prefix+@pattern].collect{|p| p.sub(prefix, "")}
  end
  rs =  @shell.record_separator unless rs
  for f in @files
    yield f+rs
  end
end
