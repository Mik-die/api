def test_push_followed_by_save_does_not_load_target
  david = authors(:david)

  david.posts << (post = Post.new(:title => "New on Edge", :body => "More cool stuff!"))
  assert !david.posts.loaded?
  david.save
  assert !david.posts.loaded?
  assert david.posts.include?(post)
end
