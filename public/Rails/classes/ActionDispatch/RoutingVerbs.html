def get(uri_or_host, path = nil, port = nil)
  host = uri_or_host.host unless path
  path ||= uri_or_host.path

  params = {'PATH_INFO'      => path,
            'REQUEST_METHOD' => 'GET',
            'HTTP_HOST'      => host}

  routes.call(params)[2].join
end
